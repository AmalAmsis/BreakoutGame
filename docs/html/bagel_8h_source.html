53 template <class T, int N>
54 class DynamicBag : NoCopy
55 {
56 public:
57 void push(const T& t) {
58 if (_size == _capacity) {
59 _capacity *= 2;
60 _arr = static_cast<T*>(
61 realloc(_arr, sizeof(T)*_capacity));
62 }
63 _arr[_size] = t;
64 ++_size;
65 }
66 void ensure(size_type s) {
67 if (_capacity < s) {
68 _capacity = std::max(s, _capacity*2);
69 _arr = static_cast<T*>(
70 realloc(_arr, sizeof(T)*_capacity));
71 }
72 }
73 T pop() { return _arr[--_size]; }
74 T& operator[](index_type i) { return _arr[i]; }
75 const T& operator[](index_type i) const { return _arr[i]; }
76 void clear() { _size = 0; }
77
78 size_type size() const { return _size; }
79 size_type capacity() const { return _capacity; }
80
81 ~DynamicBag() { free(_arr); }
82 private:
83 T* _arr = static_cast<T*>(malloc(sizeof(T) * N));
84 size_type _size = 0;
85 size_type _capacity = N;
86 };

128 class PackedStorage final : NoInstance
129 {
130 public:
131 static void add(ent_type e, const T& t) {
132 _entToComp.ensure(e.id);
133 _entToComp[e.id] = _comps.size();
134 _comps.push(t);
135 _compToEnt.push(e);
136 }
137 static void del(ent_type e) {
138 index_type ent_comp_idx = _entToComp[e.id];
139 ent_type last_ent = _compToEnt.pop();
140
141 _comps[ent_comp_idx] = _comps.pop();
142 _compToEnt[ent_comp_idx] = last_ent;
143 _entToComp[last_ent.id] = ent_comp_idx;
144 }
145 static T& get(ent_type e) {
146 return _comps[_entToComp[e.id]];
147 }
148 static int size() { return _comps.size(); }
149 static T& get(index_type idx) {
150 return _comps[idx];
151 }
152 static ent_type entity(index_type idx) {
153 return _compToEnt[idx];
154 }

197 class MultiMask final
198 {
199 public:
200 using bit_type = struct {
201 const index_type index;
202 const mask_type mask;
203 };
204 static constexpr bit_type bit(index_type idx) {
205 return {idx/BitsetWidth, static_cast<mask_type>(1<<(idx%BitsetWidth))};
206 }
207
208 void set(const bit_type& b) { _masks[b.index] |= b.mask; }
209
210 void clear(const bit_type& b) { _masks[b.index] &= ~b.mask; }
211 void clear() { memset(_masks, 0, sizeof(_masks)); }
212
213 bool test(const bit_type& b) const { return _masks[b.index] & b.mask; }
214 bool test(const MultiMask& m) const {
215 for (index_type i = 0; i < Size; ++i)
216 if ((_masks[i] & m._masks[i]) != m._masks[i])
217 return false;
218 return true;
219 }
220
221 index_type ctz() const {
222 for (index_type i = 0; i < Size; ++i) {
223 if (_masks[i]) {
224 int c = __builtin_ctz(_masks[i]);
225 return c + i*BitsetWidth;
226 }
227 }
228 return -1;
229 }
230 private:
231 static constexpr size_type Size = (Params.MaxComponents-1)/BitsetWidth + 1;
232 mask_type _masks[Size] ={};
233 };

259 static void destroyEntity(ent_type ent) {
260 if constexpr (Params.CallbackOnDestroy) {
261 Mask m = _masks[ent.id];
262 int ctz = m.ctz(); // count-trailing-zeros
263 while (ctz >= 0) {
264 if (_callbacks[ctz].destroy != nullptr)
265 _callbacks->destroy(ent);
266 m.clear(Mask::bit(ctz));
267 ctz = m.ctz();
268 }
269 }
270 _masks[ent.id].clear();
271 _ids.push(ent);
272 }
